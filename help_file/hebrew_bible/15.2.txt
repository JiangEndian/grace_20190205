15 2-1 וְאֵלֶּה בְּנֵי הַמְּדִינָה הָעֹלִים מִשְּׁבִי הַגּוֹלָה אֲשֶׁר הֶגְלָה נְבוּכַדְנֶצּוֹר נְבוּכַדְנֶצַּר מֶלֶךְ־בָּבֶל לְבָבֶל וַיָּשׁוּבוּ לִירוּשָׁלִַם וִיהוּדָה אִישׁ לְעִירוֹ:

15 2-2 אֲשֶׁר־בָּאוּ עִם־זְרֻבָּבֶל יֵשׁוּעַ נְחֶמְיָה שְׂרָיָה רְעֵלָיָה מָרְדֳּכַי בִּלְשָׁן מִסְפָּר בִּגְוַי רְחוּם בַּעֲנָה מִסְפַּר אַנְשֵׁי עַם יִשְׂרָאֵל: ס

15 2-3 בְּנֵי פַרְעֹשׁ אַלְפַּיִם מֵאָה שִׁבְעִים וּשְׁנָיִם: ס

15 2-4 בְּנֵי שְׁפַטְיָה שְׁלֹשׁ מֵאוֹת שִׁבְעִים וּשְׁנָיִם: ס

15 2-5 בְּנֵי אָרַח שְׁבַע מֵאוֹת חֲמִשָּׁה וְשִׁבְעִים: ס

15 2-6 בְּנֵי־פַחַת מוֹאָב לִבְנֵי יֵשׁוּעַ יוֹאָב אַלְפַּיִם שְׁמֹנֶה מֵאוֹת וּשְׁנֵים עָשָׂר: ס

15 2-7 בְּנֵי עֵילָם אֶלֶף מָאתַיִם חֲמִשִּׁים וְאַרְבָּעָה: ס

15 2-8 בְּנֵי זַתּוּא תְּשַׁע מֵאוֹת וְאַרְבָּעִים וַחֲמִשָּׁה: ס

15 2-9 בְּנֵי זַכָּי שְׁבַע מֵאוֹת וְשִׁשִּׁים: ס

15 2-10 בְּנֵי בָנִי שֵׁשׁ מֵאוֹת אַרְבָּעִים וּשְׁנָיִם: ס

15 2-11 בְּנֵי בֵבָי שֵׁשׁ מֵאוֹת עֶשְׂרִים וּשְׁלֹשָׁה: ס

15 2-12 בְּנֵי עַזְגָּד אֶלֶף מָאתַיִם עֶשְׂרִים וּשְׁנָיִם: ס

15 2-13 בְּנֵי אֲדֹנִיקָם שֵׁשׁ מֵאוֹת שִׁשִּׁים וְשִׁשָּׁה: ס

15 2-14 בְּנֵי בִגְוָי אַלְפַּיִם חֲמִשִּׁים וְשִׁשָּׁה: ס

15 2-15 בְּנֵי עָדִין אַרְבַּע מֵאוֹת חֲמִשִּׁים וְאַרְבָּעָה: ס

15 2-16 בְּנֵי־אָטֵר לִיחִזְקִיָּה תִּשְׁעִים וּשְׁמֹנָה: ס

15 2-17 בְּנֵי בֵצָי שְׁלֹשׁ מֵאוֹת עֶשְׂרִים וּשְׁלֹשָׁה: ס

15 2-18 בְּנֵי יוֹרָה מֵאָה וּשְׁנֵים עָשָׂר: ס

15 2-19 בְּנֵי חָשֻׁם מָאתַיִם עֶשְׂרִים וּשְׁלֹשָׁה: ס

15 2-20 בְּנֵי גִבָּר תִּשְׁעִים וַחֲמִשָּׁה: ס

15 2-21 בְּנֵי בֵית־לָחֶם מֵאָה עֶשְׂרִים וּשְׁלֹשָׁה: ס

15 2-22 אַנְשֵׁי נְטֹפָה חֲמִשִּׁים וְשִׁשָּׁה:

15 2-23 אַנְשֵׁי עֲנָתוֹת מֵאָה עֶשְׂרִים וּשְׁמֹנָה: ס

15 2-24 בְּנֵי עַזְמָוֶת אַרְבָּעִים וּשְׁנָיִם: ס

15 2-25 בְּנֵי קִרְיַת עָרִים כְּפִירָה וּבְאֵרוֹת שְׁבַע מֵאוֹת וְאַרְבָּעִים וּשְׁלֹשָׁה: ס

15 2-26 בְּנֵי הָרָמָה וָגָבַע שֵׁשׁ מֵאוֹת עֶשְׂרִים וְאֶחָד: ס

15 2-27 אַנְשֵׁי מִכְמָס מֵאָה עֶשְׂרִים וּשְׁנָיִם: ס

15 2-28 אַנְשֵׁי בֵית־אֵל וְהָעָי מָאתַיִם עֶשְׂרִים וּשְׁלֹשָׁה: ס

15 2-29 בְּנֵי נְבוֹ חֲמִשִּׁים וּשְׁנָיִם: ס

15 2-30 בְּנֵי מַגְבִּישׁ מֵאָה חֲמִשִּׁים וְשִׁשָּׁה: ס

15 2-31 בְּנֵי עֵילָם אַחֵר אֶלֶף מָאתַיִם חֲמִשִּׁים וְאַרְבָּעָה: ס

15 2-32 בְּנֵי חָרִם שְׁלֹשׁ מֵאוֹת וְעֶשְׂרִים: ס

15 2-33 בְּנֵי־לֹד חָדִיד וְאוֹנוֹ שְׁבַע מֵאוֹת עֶשְׂרִים וַחֲמִשָּׁה: ס

15 2-34 בְּנֵי יְרֵחוֹ שְׁלֹשׁ מֵאוֹת אַרְבָּעִים וַחֲמִשָּׁה: ס

15 2-35 בְּנֵי סְנָאָה שְׁלֹשֶׁת אֲלָפִים וְשֵׁשׁ מֵאוֹת וּשְׁלֹשִׁים: ס

15 2-36 הַכֹּהֲנִים בְּנֵי יְדַעְיָה לְבֵית יֵשׁוּעַ תְּשַׁע מֵאוֹת שִׁבְעִים וּשְׁלֹשָׁה: ס

15 2-37 בְּנֵי אִמֵּר אֶלֶף חֲמִשִּׁים וּשְׁנָיִם: ס

15 2-38 בְּנֵי פַשְׁחוּר אֶלֶף מָאתַיִם אַרְבָּעִים וְשִׁבְעָה: ס

15 2-39 בְּנֵי חָרִם אֶלֶף וְשִׁבְעָה עָשָׂר: ס

15 2-40 הַלְוִיִּם בְּנֵי־יֵשׁוּעַ וְקַדְמִיאֵל לִבְנֵי הוֹדַוְיָה שִׁבְעִים וְאַרְבָּעָה: ס

15 2-41 הַמְשֹׁרְרִים בְּנֵי אָסָף מֵאָה עֶשְׂרִים וּשְׁמֹנָה: פ

15 2-42 בְּנֵי הַשֹּׁעֲרִים בְּנֵי־שַׁלּוּם בְּנֵי־אָטֵר בְּנֵי־טַלְמוֹן בְּנֵי־עַקּוּב בְּנֵי חֲטִיטָא בְּנֵי שֹׁבָי הַכֹּל מֵאָה שְׁלֹשִׁים וְתִשְׁעָה: פ

15 2-43 הַנְּתִינִים בְּנֵי־צִיחָא בְנֵי־חֲשׂוּפָא בְּנֵי טַבָּעוֹת:

15 2-44 בְּנֵי־קֵרֹס בְּנֵי־סִיעֲהָא בְּנֵי פָדוֹן:

15 2-45 בְּנֵי־לְבָנָה בְנֵי־חֲגָבָה בְּנֵי עַקּוּב:

15 2-46 בְּנֵי־חָגָב בְּנֵי־שַׁמְלַי שַׁלְמַי בְּנֵי חָנָן:

15 2-47 בְּנֵי־גִדֵּל בְּנֵי־גַחַר בְּנֵי רְאָיָה:

15 2-48 בְּנֵי־רְצִין בְּנֵי־נְקוֹדָא בְּנֵי גַזָּם:

15 2-49 בְּנֵי־עֻזָּא בְנֵי־פָסֵחַ בְּנֵי בֵסָי:

15 2-50 בְּנֵי־אַסְנָה בְנֵי־מְעִינִים מְעוּנִים בְּנֵי נְפִיסִים נְפוּסִים:

15 2-51 בְּנֵי־בַקְבּוּק בְּנֵי־חֲקוּפָא בְּנֵי חַרְחוּר:

15 2-52 בְּנֵי־בַצְלוּת בְּנֵי־מְחִידָא בְּנֵי חַרְשָׁא:

15 2-53 בְּנֵי־בַרְקוֹס בְּנֵי־סִיסְרָא בְּנֵי־תָמַח:

15 2-54 בְּנֵי נְצִיחַ בְּנֵי חֲטִיפָא:

15 2-55 בְּנֵי עַבְדֵי שְׁלֹמֹה בְּנֵי־סֹטַי בְּנֵי־הַסֹּפֶרֶת בְּנֵי פְרוּדָא:

15 2-56 בְּנֵי־יַעְלָה בְנֵי־דַרְקוֹן בְּנֵי גִדֵּל:

15 2-57 בְּנֵי שְׁפַטְיָה בְנֵי־חַטִּיל בְּנֵי פֹּכֶרֶת הַצְּבָיִים בְּנֵי אָמִי:

15 2-58 כָּל־הַנְּתִינִים וּבְנֵי עַבְדֵי שְׁלֹמֹה שְׁלֹשׁ מֵאוֹת תִּשְׁעִים וּשְׁנָיִם: ס

15 2-59 וְאֵלֶּה הָעֹלִים מִתֵּל מֶלַח תֵּל חַרְשָׁא כְּרוּב אַדָּן אִמֵּר וְלֹא יָכְלוּ לְהַגִּיד בֵּית־אֲבוֹתָם וְזַרְעָם אִם מִיִּשְׂרָאֵל הֵם:

15 2-60 בְּנֵי־דְלָיָה בְנֵי־טוֹבִיָּה בְּנֵי נְקוֹדָא שֵׁשׁ מֵאוֹת חֲמִשִּׁים וּשְׁנָיִם: ס

15 2-61 וּמִבְּנֵי הַכֹּהֲנִים בְּנֵי חֳבַיָּה בְּנֵי הַקֹּוץ בְּנֵי בַרְזִלַּי אֲשֶׁר לָקַח מִבְּנוֹת בַּרְזִלַּי הַגִּלְעָדִי אִשָּׁה וַיִּקָּרֵא עַל־שְׁמָם:

15 2-62 אֵלֶּה בִּקְשׁוּ כְתָבָם הַמִּתְיַחְשִׂים וְלֹא נִמְצָאוּ וַיְגֹאֲלוּ מִן־הַכְּהֻנָּה:

15 2-63 וַיֹּאמֶר הַתִּרְשָׁתָא לָהֶם אֲשֶׁר לֹא־יֹאכְלוּ מִקֹּדֶשׁ הַקֳּדָשִׁים עַד עֲמֹד כֹּהֵן לְאוּרִים וּלְתֻמִּים:

15 2-64 כָּל־הַקָּהָל כְּאֶחָד אַרְבַּע רִבּוֹא אַלְפַּיִם שְׁלֹשׁ־מֵאוֹת שִׁשִּׁים:

15 2-65 מִלְּבַד עַבְדֵיהֶם וְאַמְהֹתֵיהֶם אֵלֶּה שִׁבְעַת אֲלָפִים שְׁלֹשׁ מֵאוֹת שְׁלֹשִׁים וְשִׁבְעָה וְלָהֶם מְשֹׁרְרִים וּמְשֹׁרְרוֹת מָאתָיִם:

15 2-66 סוּסֵיהֶם שְׁבַע מֵאוֹת שְׁלֹשִׁים וְשִׁשָּׁה פִּרְדֵיהֶם מָאתַיִם אַרְבָּעִים וַחֲמִשָּׁה:

15 2-67 גְּמַלֵּיהֶם אַרְבַּע מֵאוֹת שְׁלֹשִׁים וַחֲמִשָּׁה חֲמֹרִים שֵׁשֶׁת אֲלָפִים שְׁבַע מֵאוֹת וְעֶשְׂרִים: פ

15 2-68 וּמֵרָאשֵׁי הָאָבוֹת בְּבוֹאָם לְבֵית יְהוָה אֲשֶׁר בִּירוּשָׁלָםִ הִתְנַדְּבוּ לְבֵית הָאֱלֹהִים לְהַעֲמִידוֹ עַל־מְכוֹנוֹ:

15 2-69 כְּכֹחָם נָתְנוּ לְאוֹצַר הַמְּלָאכָה זָהָב דַּרְכְּמוֹנִים שֵׁשׁ־רִבֹּאות וָאֶלֶף ס וְכֶסֶף מָנִים חֲמֵשֶׁת אֲלָפִים וְכָתְנֹת כֹּהֲנִים מֵאָה: ס

15 2-70 וַיֵּשְׁבוּ הַכֹּהֲנִים וְהַלְוִיִּם וּמִן־הָעָם וְהַמְשֹׁרְרִים וְהַשּׁוֹעֲרִים וְהַנְּתִינִים בְּעָרֵיהֶם וְכָל־יִשְׂרָאֵל בְּעָרֵיהֶם: ס
