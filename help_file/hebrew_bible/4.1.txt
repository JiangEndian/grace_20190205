4 1-1 וַיְדַבֵּר יְהוָה אֶל־מֹשֶׁה בְּמִדְבַּר סִינַי בְּאֹהֶל מוֹעֵד בְּאֶחָד לַחֹדֶשׁ הַשֵּׁנִי בַּשָּׁנָה הַשֵּׁנִית לְצֵאתָם מֵאֶרֶץ מִצְרַיִם לֵאמֹר:

4 1-2 שְׂאוּ אֶת־רֹאשׁ כָּל־עֲדַת בְּנֵי־יִשְׂרָאֵל לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמוֹת כָּל־זָכָר לְגֻלְגְּלֹתָם:

4 1-3 מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כָּל־יֹצֵא צָבָא בְּיִשְׂרָאֵל תִּפְקְדוּ אֹתָם לְצִבְאֹתָם אַתָּה וְאַהֲרֹן:

4 1-4 וְאִתְּכֶם יִהְיוּ אִישׁ אִישׁ לַמַּטֶּה אִישׁ רֹאשׁ לְבֵית־אֲבֹתָיו הוּא:

4 1-5 וְאֵלֶּה שְׁמוֹת הָאֲנָשִׁים אֲשֶׁר יַעַמְדוּ אִתְּכֶם לִרְאוּבֵן אֱלִיצוּר בֶּן־שְׁדֵיאוּר:

4 1-6 לְשִׁמְעוֹן שְׁלֻמִיאֵל בֶּן־צוּרִישַׁדָּי:

4 1-7 לִיהוּדָה נַחְשׁוֹן בֶּן־עַמִּינָדָב:

4 1-8 לְיִשָּׂשכָר נְתַנְאֵל בֶּן־צוּעָר:

4 1-9 לִזְבוּלֻן אֱלִיאָב בֶּן־חֵלֹן:

4 1-10 לִבְנֵי יוֹסֵף לְאֶפְרַיִם אֱלִישָׁמָע בֶּן־עַמִּיהוּד לִמְנַשֶּׁה גַּמְלִיאֵל בֶּן־פְּדָהצוּר:

4 1-11 לְבִנְיָמִן אֲבִידָן בֶּן־גִּדְעֹנִי:

4 1-12 לְדָן אֲחִיעֶזֶר בֶּן־עַמִּישַׁדָּי:

4 1-13 לְאָשֵׁר פַּגְעִיאֵל בֶּן־עָכְרָן:

4 1-14 לְגָד אֶלְיָסָף בֶּן־דְּעוּאֵל:

4 1-15 לְנַפְתָּלִי אֲחִירַע בֶּן־עֵינָן:

4 1-16 אֵלֶּה קְרִיאֵי קְרוּאֵי הָעֵדָה נְשִׂיאֵי מַטּוֹת אֲבוֹתָם רָאשֵׁי אַלְפֵי יִשְׂרָאֵל הֵם:

4 1-17 וַיִּקַּח מֹשֶׁה וְאַהֲרֹן אֵת הָאֲנָשִׁים הָאֵלֶּה אֲשֶׁר נִקְּבוּ בְּשֵׁמוֹת:

4 1-18 וְאֵת כָּל־הָעֵדָה הִקְהִילוּ בְּאֶחָד לַחֹדֶשׁ הַשֵּׁנִי וַיִּתְיַלְדוּ עַל־מִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמוֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה לְגֻלְגְּלֹתָם:

4 1-19 כַּאֲשֶׁר צִוָּה יְהוָה אֶת־מֹשֶׁה וַיִּפְקְדֵם בְּמִדְבַּר סִינָי: פ

4 1-20 וַיִּהְיוּ בְנֵי־רְאוּבֵן בְּכֹר יִשְׂרָאֵל תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמוֹת לְגֻלְגְּלֹתָם כָּל־זָכָר מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-21 פְּקֻדֵיהֶם לְמַטֵּה רְאוּבֵן שִׁשָּׁה וְאַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת: פ

4 1-22 לִבְנֵי שִׁמְעוֹן תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם פְּקֻדָיו בְּמִסְפַּר שֵׁמוֹת לְגֻלְגְּלֹתָם כָּל־זָכָר מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-23 פְּקֻדֵיהֶם לְמַטֵּה שִׁמְעוֹן תִּשְׁעָה וַחֲמִשִּׁים אֶלֶף וּשְׁלֹשׁ מֵאוֹת: פ

4 1-24 לִבְנֵי גָד תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמוֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-25 פְּקֻדֵיהֶם לְמַטֵּה גָד חֲמִשָּׁה וְאַרְבָּעִים אֶלֶף וְשֵׁשׁ מֵאוֹת וַחֲמִשִּׁים: פ

4 1-26 לִבְנֵי יְהוּדָה תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-27 פְּקֻדֵיהֶם לְמַטֵּה יְהוּדָה אַרְבָּעָה וְשִׁבְעִים אֶלֶף וְשֵׁשׁ מֵאוֹת: פ

4 1-28 לִבְנֵי יִשָּׂשכָר תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-29 פְּקֻדֵיהֶם לְמַטֵּה יִשָּׂשכָר אַרְבָּעָה וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת: פ

4 1-30 לִבְנֵי זְבוּלֻן תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-31 פְּקֻדֵיהֶם לְמַטֵּה זְבוּלֻן שִׁבְעָה וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת: פ

4 1-32 לִבְנֵי יוֹסֵף לִבְנֵי אֶפְרַיִם תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-33 פְּקֻדֵיהֶם לְמַטֵּה אֶפְרָיִם אַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת: פ

4 1-34 לִבְנֵי מְנַשֶּׁה תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמוֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-35 פְּקֻדֵיהֶם לְמַטֵּה מְנַשֶּׁה שְׁנַיִם וּשְׁלֹשִׁים אֶלֶף וּמָאתָיִם: פ

4 1-36 לִבְנֵי בִנְיָמִן תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-37 פְּקֻדֵיהֶם לְמַטֵּה בִנְיָמִן חֲמִשָּׁה וּשְׁלֹשִׁים אֶלֶף וְאַרְבַּע מֵאוֹת: פ

4 1-38 לִבְנֵי דָן תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-39 פְּקֻדֵיהֶם לְמַטֵּה דָן שְׁנַיִם וְשִׁשִּׁים אֶלֶף וּשְׁבַע מֵאוֹת: פ

4 1-40 לִבְנֵי אָשֵׁר תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-41 פְּקֻדֵיהֶם לְמַטֵּה אָשֵׁר אֶחָד וְאַרְבָּעִים אֶלֶף וַחֲמֵשׁ מֵאוֹת: פ

4 1-42 בְּנֵי נַפְתָּלִי תּוֹלְדֹתָם לְמִשְׁפְּחֹתָם לְבֵית אֲבֹתָם בְּמִסְפַּר שֵׁמֹת מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כֹּל יֹצֵא צָבָא:

4 1-43 פְּקֻדֵיהֶם לְמַטֵּה נַפְתָּלִי שְׁלֹשָׁה וַחֲמִשִּׁים אֶלֶף וְאַרְבַּע מֵאוֹת: פ

4 1-44 אֵלֶּה הַפְּקֻדִים אֲשֶׁר פָּקַד מֹשֶׁה וְאַהֲרֹן וּנְשִׂיאֵי יִשְׂרָאֵל שְׁנֵים עָשָׂר אִישׁ אִישׁ־אֶחָד לְבֵית־אֲבֹתָיו הָיוּ:

4 1-45 וַיִּהְיוּ כָּל־פְּקוּדֵי בְנֵי־יִשְׂרָאֵל לְבֵית אֲבֹתָם מִבֶּן עֶשְׂרִים שָׁנָה וָמַעְלָה כָּל־יֹצֵא צָבָא בְּיִשְׂרָאֵל:

4 1-46 וַיִּהְיוּ כָּל־הַפְּקֻדִים שֵׁשׁ־מֵאוֹת אֶלֶף וּשְׁלֹשֶׁת אֲלָפִים וַחֲמֵשׁ מֵאוֹת וַחֲמִשִּׁים:

4 1-47 וְהַלְוִיִּם לְמַטֵּה אֲבֹתָם לֹא הָתְפָּקְדוּ בְּתוֹכָם: פ

4 1-48 וַיְדַבֵּר יְהוָה אֶל־מֹשֶׁה לֵּאמֹר:

4 1-49 אַךְ אֶת־מַטֵּה לֵוִי לֹא תִפְקֹד וְאֶת־רֹאשָׁם לֹא תִשָּׂא בְּתוֹךְ בְּנֵי יִשְׂרָאֵל:

4 1-50 וְאַתָּה הַפְקֵד אֶת־הַלְוִיִּם עַל־מִשְׁכַּן הָעֵדֻת וְעַל כָּל־כֵּלָיו וְעַל כָּל־אֲשֶׁר־לוֹ הֵמָּה יִשְׂאוּ אֶת־הַמִּשְׁכָּן וְאֶת־כָּל־כֵּלָיו וְהֵם יְשָׁרְתֻהוּ וְסָבִיב לַמִּשְׁכָּן יַחֲנוּ:

4 1-51 וּבִנְסֹעַ הַמִּשְׁכָּן יוֹרִידוּ אֹתוֹ הַלְוִיִּם וּבַחֲנֹת הַמִּשְׁכָּן יָקִימוּ אֹתוֹ הַלְוִיִּם וְהַזָּר הַקָּרֵב יוּמָת:

4 1-52 וְחָנוּ בְּנֵי יִשְׂרָאֵל אִישׁ עַל־מַחֲנֵהוּ וְאִישׁ עַל־דִּגְלוֹ לְצִבְאֹתָם:

4 1-53 וְהַלְוִיִּם יַחֲנוּ סָבִיב לְמִשְׁכַּן הָעֵדֻת וְלֹא־יִהְיֶה קֶצֶף עַל־עֲדַת בְּנֵי יִשְׂרָאֵל וְשָׁמְרוּ הַלְוִיִּם אֶת־מִשְׁמֶרֶת מִשְׁכַּן הָעֵדוּת:

4 1-54 וַיַּעֲשׂוּ בְּנֵי יִשְׂרָאֵל כְּכֹל אֲשֶׁר צִוָּה יְהוָה אֶת־מֹשֶׁה כֵּן עָשׂוּ: פ
